Creating the DataBunch¬∂

[3]:

def iris():
    "Loads the IRIS dataset into two tensors."
    iris = load_iris()
    x = torch.tensor(iris.data)
    y = torch.tensor(iris.target)
    labels = iris.feature_names

    return x, y, labels

[4]:

x, y, labels = iris()

[5]:

db = UnsupervisedDataBunch(x, valid=0.05, sampler='shuffle', bs=50)
